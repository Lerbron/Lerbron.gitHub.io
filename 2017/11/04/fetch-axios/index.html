异步请求
1、fetch与promise

const baseUrl = 'https://cnodejs.org',
    // 自定义请求头
    headers = {
          'Content-Type': 'application/json',
      'Access-Control-Allow-Origin': '*',
      'Access-Control-Allow-Methods': 'GET, POST, PUT, OPTIONS',
        'Access-Control-Allow-Headers': 'X-Requested-With',
        'Accept': '*'
    };

    const get = (url, body, responseType = 'json') => {
              url = absoluteUrl(url);

              // 返回一个自执行的promise，并将返回的数据进行json格式化
              return Promise.resolve( function () {
                  return window.fetch ?
                    fetch(url, {method: 'GET'}).then(res => res[responseType]()) :
                    Reqwest({   // 对fetch作兼容
                      url,
                      metods: 'GET'
                   })
              }())
        };

        const post = ( (url, body = {}, responseType = 'json' ) => {
            url =absoluteUrl(url);

            return Promise.resolve( function () {
            return window.fetch ?
              fetch(url, {
                  method: "POST",
                  headers,
                  body: JSON.stringify(body)
                }
              ).then( res => res[responseType]()) :
              Reqwest({     // 对fetch作兼容
                url,
                headers,
                data: JSON.stringify(body)
              })
            }())
        } );

    let globalQueryParams = {};   // 全局参数对象

    const absoluteUrl = url  => {
          let globalParams = '';

          // 格式为全局参数
          for (key in globalQueryParams) {
              globalParams += `${key}=${globalQueryParams[key]}&`
      }

      // 判断请求的接口中是否含有？，即get请求时会将参数以 ?##=## 的格式进行拼接在后面形成url
      let join = /\?/.test(url) ? '&' : '?';

      return `${baseUrl}/${url}${join}${globalParams}`; // 拼接url
    };
